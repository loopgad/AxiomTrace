# AxiomTrace Minimalist Architecture Analysis

## Triple-Dimension Conclusions

### 1. Steps That Should Be Cut Immediately

| Priority | Content to Remove | Reasoning |
|----------|-------------------|-----------|
| **P0** | YAML Dictionary as Firmware Build Dependency | Increases build complexity, requires Python environment, disconnected from MCU dev environment. |
| **P0** | `axiom_events_generated.h` Code Generation | Can be entirely handled by C macros at compile time, no external tools needed. |
| **P1** | Split into 7 Header + 7 C Files | Over-engineered; for v0.1, it should be merged into a minimal set. |
| **P1** | CMake as the Sole Build Entry | Should support direct copying of .c/.h into existing projects for compilation. |

**Core Judgment**: The current design follows a "host-side toolchain first" mindset, but the real pain point for MCU developers is "I don't want to install Python."

---

### 2. Can We Achieve "Zero YAML, Zero External Tools, Zero Generated Files"?

**Conclusion: Yes, but with trade-offs.**

#### Option A: Pure X-Macro Source of Truth (Recommended as Default Layer)

```c
/* axiom_events.h —— The only event definition file developers need to edit */

#define AXIOM_EVENTS_H
```

```c
/* axiom.h —— Core header file, containing all code */
#ifndef AXIOM_H
#define AXIOM_H

#include <stdint.h>
#include <stddef.h>

/* Users must define the event list before including this header */
#ifndef AXIOM_MODULE_LIST
#error "Define AXIOM_MODULE_LIST before including axiom.h"
#endif
#ifndef AXIOM_EVENT_LIST
#error "Define AXIOM_EVENT_LIST before including axiom.h"
#endif

/* ---- Module ID macros automatically generated by X-Macro ---- */
#define _AXIOM_EXPAND_MODULE(name, id)  \
    static const uint8_t _AXIOM_MODULE_ID_##name = (id);

AXIOM_MODULE_LIST(_AXIOM_EXPAND_MODULE)

/* ---- Event ID macros automatically generated by X-Macro ---- */
#define _AXIOM_EXPAND_EVENT(mod, name, id, level, argc)  \
    static const uint16_t _AXIOM_EVENT_ID_##mod##_##name = (id);

AXIOM_EVENT_LIST(_AXIOM_EXPAND_EVENT)

/* ---- Log Levels ---- */
typedef enum {
    AXIOM_LEVEL_DEBUG = 0,
    AXIOM_LEVEL_INFO  = 1,
    AXIOM_LEVEL_WARN  = 2,
    AXIOM_LEVEL_ERROR = 3,
    AXIOM_LEVEL_FAULT = 4,
} axiom_level_t;

/* ---- Type-Safe Encoding (C11 _Generic) ---- */
static inline void _axiom_enc_u8 (uint8_t *b, uint8_t *p, uint8_t  v) { b[(*p)++] = 0x01; b[(*p)++] = v; }
static inline void _axiom_enc_i8 (uint8_t *b, uint8_t *p, int8_t   v) { b[(*p)++] = 0x02; b[(*p)++] = (uint8_t)v; }
static inline void _axiom_enc_u16(uint8_t *b, uint8_t *p, uint16_t v) { b[(*p)++] = 0x03; b[(*p)++] = v&0xFF; b[(*p)++] = v>>8; }
static inline void _axiom_enc_i16(uint8_t *b, uint8_t *p, int16_t  v) { b[(*p)++] = 0x04; b[(*p)++] = v&0xFF; b[(*p)++] = v>>8; }
static inline void _axiom_enc_u32(uint8_t *b, uint8_t *p, uint32_t v) { b[(*p)++] = 0x05; b[(*p)++] = v&0xFF; b[(*p)++] = (v>>8)&0xFF; b[(*p)++] = (v>>16)&0xFF; b[(*p)++] = v>>24; }
static inline void _axiom_enc_i32(uint8_t *b, uint8_t *p, int32_t  v) { b[(*p)++] = 0x06; /* ... */ }
static inline void _axiom_enc_f32(uint8_t *b, uint8_t *p, float    v) { b[(*p)++] = 0x07; /* memcpy */ }

#define _AXIOM_ENCODE_ONE(buf, pos, arg) \
    _Generic((arg), \
        uint8_t:  _axiom_enc_u8,  \
        int8_t:   _axiom_enc_i8,  \
        uint16_t: _axiom_enc_u16, \
        int16_t:  _axiom_enc_i16, \
        uint32_t: _axiom_enc_u32, \
        int32_t:  _axiom_enc_i32, \
        float:    _axiom_enc_f32  \
    )(buf, &(pos), arg)

/* ---- Variadic Argument Counting Macros ---- */
#define _AXIOM_NARGS(...) _AXIOM_NARGS_IMPL(__VA_ARGS__, 8,7,6,5,4,3,2,1,0)
#define _AXIOM_NARGS_IMPL(_1,_2,_3,_4,_5,_6,_7,_8,N,...) N
#define _AXIOM_CAT(a,b) a##b

/* ---- Log Macros Expanded by Argument Count ---- */
#define _AXIOM_LOG_0(lvl, mod, evt) \
    axiom_write((lvl), _AXIOM_MODULE_ID_##mod, _AXIOM_EVENT_ID_##mod##_##evt, NULL, 0)

#define _AXIOM_LOG_1(lvl, mod, evt, a1) \
    do{ uint8_t _b[32]; uint8_t _p=0; _AXIOM_ENCODE_ONE(_b,_p,a1); \
        axiom_write((lvl), _AXIOM_MODULE_ID_##mod, _AXIOM_EVENT_ID_##mod##_##evt, _b, _p); }while(0)

/* ... _LOG_2 to _LOG_8 omitted similarly ... */

#define _AXIOM_LOG_DISPATCH(lvl, mod, evt, ...) \
    _AXIOM_CAT(_AXIOM_LOG_, _AXIOM_NARGS(__VA_ARGS__))(lvl, mod, evt, __VA_ARGS__)

#define AXIOM_DEBUG(mod,evt,...) _AXIOM_LOG_DISPATCH(AXIOM_LEVEL_DEBUG, mod, evt, __VA_ARGS__)
#define AXIOM_INFO(mod,evt,...)  _AXIOM_LOG_DISPATCH(AXIOM_LEVEL_INFO,  mod, evt, __VA_ARGS__)
#define AXIOM_WARN(mod,evt,...)  _AXIOM_LOG_DISPATCH(AXIOM_LEVEL_WARN,  mod, evt, __VA_ARGS__)
#define AXIOM_ERROR(mod,evt,...) _AXIOM_LOG_DISPATCH(AXIOM_LEVEL_ERROR, mod, evt, __VA_ARGS__)

__attribute__((weak)) void axiom_critical_exit(void) { }
```

```c
/* axiom.c —— Single C file */
#include "axiom.h"

#ifndef AXIOM_RING_SIZE
#define AXIOM_RING_SIZE 1024
#endif

static volatile uint8_t  _ring[AXIOM_RING_SIZE];
static volatile uint16_t _ring_head;
static volatile uint16_t _ring_tail;
static volatile uint16_t _seq;

static void _ring_put(const uint8_t *data, uint8_t len) {
    axiom_critical_enter();
    for (uint8_t i = 0; i < len; i++) {
        _ring[_ring_head] = data[i];
        _ring_head = (_ring_head + 1) % AXIOM_RING_SIZE;
        if (_ring_head == _ring_tail) {
            _ring_tail = (_ring_tail + 1) % AXIOM_RING_SIZE; /* overwrite */
        }
    }
    axiom_critical_exit();
}

void axiom_write(axiom_level_t lvl, uint8_t mod, uint16_t evt,
                 const uint8_t *payload, uint8_t len) {
    /* Direct-to-Ring (D2R) Implementation */
    axiom_critical_enter();
    uint16_t total_len = 8 + 1 + len + 2; /* Header + Len + Payload + CRC (ignoring variable TS for brevity) */
    if (_ring_has_space(total_len)) {
        uint16_t pos = _ring_acquire(total_len);
        _ring_put_byte_at(pos++, 0xA5);
        _ring_put_byte_at(pos++, 0x10);
        /* ... write other fields directly and compute incremental CRC ... */
        _ring_commit(total_len);
    }
    axiom_critical_exit();
}

/* Backend Dispatch: Simplest polling implementation */
void axiom_flush(void) {
    axiom_critical_enter();
    while (_ring_tail != _ring_head) {
        axiom_byte_out(_ring[_ring_tail]);
        _ring_tail = (_ring_tail + 1) % AXIOM_RING_SIZE;
    }
    axiom_critical_exit();
}
```

#### Cost of the Pure Macro Solution

| Aspect | Cost | Severity |
|--------|------|----------|
| **Dictionary Export** | Host-side decoders need event metadata. Pure C macros cannot self-describe text templates, requiring extra tools to extract from source. | **High** |
| **ID Conflict Detection** | Cannot detect cross-module event_id conflicts at compile time (different modules can repeat), requires link-time or external checks. | Medium |
| **Argument Type Validation** | `_Generic` can only validate C types, not whether the "developer-claimed template type" matches actual encoding. | Medium |
| **ROM Usage** | `static const` module/event IDs enter ROM, but can be optimized into immediates by the compiler. | Low |
| **Maintainability** | X-Macro syntax is not friendly to beginners; error messages can be cryptic. | Medium |

**Key Insight**: The pure C macro solution is entirely feasible on the **firmware side**, but the **host-side decoder** still needs some form of dictionary. The solution — let the dictionary be reverse-generated from firmware source, rather than forward-driving firmware compilation.

### 3. Developer's Ideal First Log Code

```c
#include "axiom.h"

/* Optional: One-line configuration */
#pragma AXIOM_MODULE(MOTOR, 0x03)
#pragma AXIOM_EVENT(MOTOR, START, INFO, "rpm={u16}")

int main(void) {
    AXIOM_LOG("MOTOR_START", INFO, "rpm=%u", 3200);  /* Ideal state */
    return 0;
}
```

**But `AXIOM_LOG("MOTOR_START", INFO, "rpm=%u", 3200)` is technically infeasible**, because:

- `"MOTOR_START"` and `"rpm=%u"` would become string constants in ROM after compilation, violating the "no strings in firmware" principle.
- Parsing format strings at runtime requires `printf`-level code volume.

**Compromise (Recommended)**:

```c
AXIOM_INFO(MOTOR, START, (uint16_t)3200);  /* Strings don't enter firmware, type-safe */
```

If a `printf`-like experience is insisted upon, the format string must be parsed at compile-time. This requires:

- Custom compiler plugins (Clang plugin) — Not universal.
- Or C++20 `consteval` — Not C11.
- Or external code generators — Violates the "zero tools" goal.

**Therefore Conclusion**: Under the constraints of C11 + zero external tools, `AXIOM_INFO(MOTOR, START, rpm)` is the simplest achievable interface. Format templates stay in the host-side dictionary.

## Progressive Complexity Layering Design

```
┌─────────────────────────────────────────────────────────────┐
│ Layer 3: Advanced Control (Optional)                        │
│  - YAML Dictionary + Python codegen                          │
│  - Fine-grained module/event management for large teams      │
│  - Generates dictionary.json for decoders                    │
├─────────────────────────────────────────────────────────────┤
│ Layer 2: Structured Definition (Recommended)               │
│  - Manually edit axiom_events.h (X-Macro)                   │
│  - Zero external dependencies for firmware compilation       │
│  - Script provided to extract dictionary from X-Macro        │
├─────────────────────────────────────────────────────────────┤
│ Layer 1: Minimalist Mode (Default)                          │
│  - 2 files: axiom.h + axiom.c                                │
│  - Directly hardcoded module_id / event_id                  │
│  - Suitable for POC, rapid validation, small projects       │
└─────────────────────────────────────────────────────────────┘
```

## Minimal File Set Argument

| Solution | File Count | Composition | Use Case |
|----------|------------|-------------|----------|
| **Extreme Minimalism** | 2 | `axiom.h` + `axiom.c` | Rapid validation, teaching, small projects. |
| **Recommended Default** | 3 | `axiom.h` + `axiom.c` + `axiom_events.h` (+ D2R) | Production projects needing module/event management. |
| **Current Design** | 14+ | 7 .h + 7 .c + various backends | Over-split, should be deferred to v0.5+. |

**3-File Solution Content Allocation**:

1. **`axiom_events.h`** —— Edited by user: Module and event definitions (X-Macro).
2. **`axiom.h`** —— Provided by library: All macros, types, inline functions, weak symbol default implementations.
3. **`axiom.c`** —— Provided by library: Ring buffer, encoding, CRC, backend dispatch.

---

## Final Suggestions

1. **Change the default path to "Zero YAML, Zero Tools"**: Use X-Macro as the source of truth; firmware compilation does not depend on Python.
2. **Reverse Dictionary Generation**: Provide an optional `extract_dict.py` to extract JSON from `axiom_events.h` for decoder use. This is a CI/release step, not a build dependency.
3. **File Merging**: For v0.1~v0.2, compress core code into `axiom.h` + `axiom.c`. Backends provided as separate files but not required.
4. **Interface Locking**: `AXIOM_INFO(MODULE, EVENT, args...)` as the stable API; internal implementation can iterate.
5. **Retain Advanced Path**: YAML + codegen as Layer 3 for large projects requiring fine-grained control.
